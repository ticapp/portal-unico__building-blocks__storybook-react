<h1 className="mb-32">
  <u>Use Outside Element Click</u>
</h1>

<h2>Description</h2>

<p className="fs-16">
  Hook to detect and execute a callback whenever the user click outside of the provided element. Must be used in conjunction with a element
  ref.
</p>

<h2>Hook usage</h2>

```tsx
import React, { useRef } from 'react';
import { useOutsideElementClick } from 'ama-react-design-system/useOutsideElementClick';

const dummyComponent = () => {
  const dummyRef = useRef(null);
  useOutsideElementClick(dummyRef, () => customCallback());

  const customCallback = () => {
    if (dummyRef) {
      (dummyRef.current as HTMLDivElement).innerHTML = 'Clicked outside of dummy component';
    }
  };

  return <div ref={dummyRef}>Dummy Component</div>;
};
```
