  <u>Use Window Size</u>
</h1>

<h2>Description</h2>

<p className="fs-16">Hook to get window size, use this when check if component render on mobile, tablet or desktop for example.</p>

<h2>Hook usage</h2>

```tsx
import { useWindowSize } from '@ama/react-design-system';

const dummyComponent = () => {
  const { width, height } = useWindowSize();

  return (
    <>
      <p>{width}</p>
      <p>{height}</p>
    </>
  );
};
```
