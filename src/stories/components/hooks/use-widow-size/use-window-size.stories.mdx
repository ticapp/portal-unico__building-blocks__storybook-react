<h2>Description</h2>

<p>Hook to get window size</p>

<h2>Hook usage</h2>

```tsx
import { useWindowSize } from '@ama/useWindowSize';

const dummyComponent = () => {
  const { width, height } = useWindowSize();

  return (
    <>
      <p>{width}</p>
      <p>{height}</p>
    </>
  );
};
```
