<h2>Description</h2>

<p>Renders Breadcrumbs component</p>

<h2>Component usage</h2>

```jsx
const args = {
  breadcrumbs: [
    {
      name: "Início",
      url: "/home",
    },
    {
      name: "Infrações",
      url: "/violations",
    },
    {
      name: "Histórico de infrações",
      url: "/iframe.html",
    },
  ],
};

return <Breadcrumb breadcrumbs={args.breadcrumbs} />;
```
