<h2>Description</h2>

<p className="fs-16">BreadCrumb allows to know the full path to certain page and navigate backwards.</p>
<p className="fs-16">Check all props available in each example on Docs tab.</p>

<h2>Component usage</h2>

```tsx
import { BreadCrumb } from '@ama/react-design-system';

const args = {
  navAriaLabel: 'Breadcrumb',
  ariaHasPopUp: true,
  linkAriaLabel: 'page',
  buttonAriaLabel: 'Open Breadcrumb',
  breadcrumbTitle: 'Você está aqui:',
  breadcrumbs: [
    {
      name: 'Início',
      url: '/home'
    },
    {
      name: 'Infrações',
      url: '/violations'
    },
    {
      name: 'Histórico de infrações',
      url: '/iframe.html'
    }
  ]
};

<BreadCrumb {...args} />;
```
