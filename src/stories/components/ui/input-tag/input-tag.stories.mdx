const options = [

        label: 'Option #1',
        id: 'GUID 1',
    },
    {
        label: 'Option #2',
        id: 'GUID 2',
    },
    {
        label: 'Option #3',
        id: 'GUID 3',
    },
    {
        label: 'Option #4',
        id: 'GUID 4',
    },
    {
        label: 'Option #5',
        id: 'GUID 5',
    },
] as InputTagOption[];

const [tags, setTags] = useState([] as InputTagOption[]);

useEffect(() => {
  console.log(tags);
}, [tags]);

<InputTag onChange={(e) => setTags(e)} options={options} />
