<h2>Component usage</h2>

```tsx
import { toast, ToastContainer } from '@ama/react-design-system';

const Example = () => {
	const onClick = () => {
		toast({ type: 'warning', text: 'Hi there!' });
	};

	return (
	<div>
		<Button onClick={onClick}>show toast</Button>
		<ToastContainer />
	</div>
);
}
```
