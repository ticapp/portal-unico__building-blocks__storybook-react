<h2>Component usage</h2>

```tsx
import { useToast, ToastContainer, ToastProps, ToastType } from '@ama/react-design-system';

const Example = () => {
	const factory = useToast();
	const onClick = () => {
		factory.useToast({ type: ToastType.WARNING, content: 'Hi there!' });
	};

	return (
	<ToastContainer>
		<Button onClick={onClick}>show toast</Button>
	</ToastContainer>
);
}
```
