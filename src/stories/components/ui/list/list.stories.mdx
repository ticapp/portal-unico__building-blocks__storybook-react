<h2>Component usage </h2>

<label>Simple list without links </label>

```tsx
import { List } from 'ama-react-design-system/list';

<List
  listData={[
    {
      value: 'Contactos ePortugal (external Link)',
      hasExternalLink: false,
      link: '',
      title: ''
    },
    {
      value: 'Componente Logo (inside link)',
      hasExternalLink: false,
      link: '',
      title: ''
    }
  ]}
></List>;
```

<label>List with/without links and custom bullets. See more info on properties. </label>

```tsx
import { List } from 'ama-react-design-system/list';

<List
  listData={[
    {
      value: 'Contactos ePortugal (external Link)',
      hasExternalLink: false,
      link: '',
      title: ''
    },
    {
      value: 'Componente Logo (inside link)',
      hasExternalLink: false,
      link: '',
      title: ''
    }
  ]}
  listStyleType="square"
  listStyleImageUrl="https://picsum.photos/14/14"
  listStylePosition="inside"
  className="own_class"
></List>;
```
