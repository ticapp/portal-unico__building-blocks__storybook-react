<h2>Component usage</h2>

```tsx
import { Select } from '@ama/react-design-system';

const options = [
  {
    label: 'Option #1',
    value: 1
  },
  {
    label: 'Option #2',
    value: 2
  },
  {
    label: 'Option #3',
    value: 3
  },
  {
    label: 'Option #4',
    value: 4
  },
  {
    label: 'Option #5',
    value: 5
  }
] as SelectOption[];

<Select options={options} />;
```
