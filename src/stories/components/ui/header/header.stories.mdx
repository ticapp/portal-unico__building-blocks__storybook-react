<h2>Component usage</h2>

<p className="fs-16">With no languages available, the language selector will be rendered with the default language.</p>

```tsx
import { Header } from '@ama/react-design-system';

const links: HorizontalMenuLink[] = [
  {
    id: '1',
    label: 'Serviços',
    link: '/services'
  },
  {
    id: '2',
    label: 'Entidades',
    link: '/entities'
  },
  {
    id: '3',
    label: 'Atendimento',
    link: '/attendance'
  },
  {
    id: '4',
    label: 'Notícias',
    link: '/news'
  }
];

const options = [
  {
    authenticatedOption: false,
    value: 'login',
    icon: 'ama-login',
    label: 'Iniciar sessão'
  },
  {
    authenticatedOption: false,
    value: 'register',
    icon: 'ama-add-user',
    label: 'Criar conta'
  },

  {
    authenticatedOption: true,
    value: 'user-area',
    icon: 'ama-my-area',
    label: 'A minha área'
  },
  {
    authenticatedOption: true,
    value: 'logout',
    icon: 'ama-exit',
    label: 'Sair'
  }
] as UserAreaOption[];

<Header isHomepage={false} homepageLink={'/'} options={options} links={links} />;
```

```tsx
import { Header } from '@ama/react-design-system';

const links: HorizontalMenuLink[] = [
  {
    id: '1',
    label: 'Serviços',
    link: '/services'
  },
  {
    id: '2',
    label: 'Entidades',
    link: '/entities'
  },
  {
    id: '3',
    label: 'Atendimento',
    link: '/attendance'
  },
  {
    id: '4',
    label: 'Notícias',
    link: '/news'
  }
];

const options = [
  {
    authenticatedOption: false,
    value: 'login',
    icon: 'ama-login',
    label: 'Iniciar sessão'
  },
  {
    authenticatedOption: false,
    value: 'register',
    icon: 'ama-add-user',
    label: 'Criar conta'
  },

  {
    authenticatedOption: true,
    value: 'user-area',
    icon: 'ama-my-area',
    label: 'A minha área'
  },
  {
    authenticatedOption: true,
    value: 'logout',
    icon: 'ama-exit',
    label: 'Sair'
  }
] as UserAreaOption[];

const languages: SelectOption[] = [
  { label: 'PT', value: 'pt' },
  { label: 'EN', value: 'en' },
  { label: 'ES', value: 'es' }
];

<Header isHomepage={false} homepageLink={'/'} languages={languages} activeLanguage={languages[0]} options={options} links={links} />;
```
