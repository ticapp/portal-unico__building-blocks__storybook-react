<h2>Component usage</h2>

```tsx
import { InfractionsChart } from '@ama-react-design-system/infraction-chart';

const counters = [
  {
    label: 'Muito Grave',
    value: 33,
    color: '#AA120A'
  },
  {
    label: 'Grave',
    value: 33,
    color: '#FBCF5F'
  },
  {
    label: 'Leve',
    value: 33,
    color: '#00672F'
  }
] as InfractionCounter[];

<InfractionsChart counters={counters} />;
```
