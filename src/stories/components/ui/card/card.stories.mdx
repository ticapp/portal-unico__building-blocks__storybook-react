<h2>Description </h2>
<p className="fs-16">
  A component that renders cards that could be clickable using{' '}
  <a href="https://getbootstrap.com/docs/5.1/helpers/stretched-link/"> Stretched link</a> by Bootstrap. Changing some of props we can change
  style of card and some elements could be activated/disabled.
</p>
<p className="fs-16">Check all props available in each example on Docs tab.</p>

<h2>Component usage</h2>

```tsx
import { Card } from '@ama-react-design-system/card';

const args = {
  mainTitle: 'Consultar pontos da carta de condução',
  description: 'Número atual de pontos da sua carta de condução',
  contentIcon: 'ama-badge',
  content: '15 pontos',
  link: 'https://www.ama.pt',
  isExternal: true,
  title: 'link para o site AMA',
  linkLabel: 'Default',
  linkIcon: 'ama-arrow-down-right',
  isLinkLabel: true,
  cardHasLink: true
};

<Card {...args} />;
```
