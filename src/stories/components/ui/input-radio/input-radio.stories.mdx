<h2>Description</h2>

<p>Render the input radio</p>

<h2>Component usage</h2>

```js
const [inputValue, setInputValue] = useState('');

const args = {
  label: 'Label',
  name: 'input-text1',
  value: '',
  inputId: uuidv4(),
  isDisabled: false
};


<InputRadio 
    value={inputValue} 
    {...args}  
    onChange={e => setInputValue(e.target.value)} 
/>


```
