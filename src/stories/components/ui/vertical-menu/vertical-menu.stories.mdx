<h2>Component usage</h2>

```tsx
import { VerticalMenu } from 'ama-react-design-system/vertical-menu';

const links: VerticalMenuLink[] = [
  {
    label: 'Lvl 1 link 1',
    link: 'https://example.com'
  },
  {
    label: 'Lvl 1 link 2',
    children: [
      {
        label: 'Lvl 2 link 1',
        link: 'https://example.com'
      },
      {
        label: 'Lvl 2 link 2',
        link: 'https://example.com'
      },
      {
        label: 'Lvl 2 link 3',
        children: [
          {
            label: 'Lvl 3 link 1',
            link: 'https://example.com'
          },
          {
            label: 'Lvl 3 link 2',
            link: 'https://example.com'
          }
        ]
      }
    ]
  },
  {
    label: 'Lvl 1 link 3',
    link: 'https://example.com'
  },
  {
    label: 'Lvl 1 link 4',
    link: 'https://example.com'
  }
];

<VerticalMenu links={links} />;
```
