<h2>Component usage</h2>

```tsx
import { Tabs } from '@ama/react-design-system';

const args = {
  'defaultActiveKey': 'tab-first',
  'id': 'basic-tabs',
  'tabsName': [
    {
      label: 'Carta de condução',
      children: (
        <div className="mt-48">
          <h3>Conduza com cuidado!</h3>
        </div>
      ),
      keyTab: 'tab-first'
    },
    {
      label: 'Perguntas frequentes',
      children: (
        <div className="mt-48">
          <h3>Encontre as suas respostas</h3>
        </div>
      ),
      keyTab: 'tab-second'
    },
    {
      label: 'Disabled tab',
      children: (
        <div className="mt-48">
          <h3>Disabled</h3>
        </div>
      ),
      keyTab: 'tab-third',
      disabled: true
    }
  ],
  'aria-label': 'Separadores do serviço carta de condução'
};

<Tabs {...args} />;
```
