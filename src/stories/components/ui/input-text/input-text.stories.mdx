<p className="fs-16">A component that renders a free input with possibility of diferent types.</p>
<p className="fs-16">Check all props available on Docs tab.</p>

<h2>Component usage</h2>

```tsx
import { InputText } from '@ama-design-system/input-text';

const [inputValue, setInputValue] = useState('');

<InputText
  label="Nome"
  placeholder="placeholder"
  value={inputValue}
  name="inputName"
  inputId="inputId_1"
  onChange={(e) => setInputValue(e.target.value)}
/>;
```
