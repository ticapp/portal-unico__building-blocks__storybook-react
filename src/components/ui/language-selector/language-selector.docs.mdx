import { LanguageSelector } from './language-selector';

<h2>Description</h2>

<p>Renders the language selector component. When it change, it will trigger the <b>onLanguageChange</b> callback</p>

<h2>Component usage</h2>

```js
<LanguageSelector />
```

```js
<LanguageSelector languages={{ pt: 'PT', en: 'EN', es: 'ES' }} active={'pt'} />
```
