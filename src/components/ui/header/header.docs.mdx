import { Header } from './header';

<h2>Description</h2>

<p>Renders the header component in its multiple variants</p>

<h2>Component usage</h2>

<p>With no languages available, the dropdown list its not rendered.</p>

```jsx
  <Header isHomepage={false} homepageLink={'/'} />
```
<p>With only a single language provided, the dropdown list will be disabled automatically.</p>

```jsx
  <Header isHomepage={false} homepageLink={'/'} languages={{pt:'PT', en: 'EN', es:'ES'}} activeLanguage={'pt'} />
```
